        Ques: Write a function that accepts parameter n and returns factorial of n
            Ans:-  function fact1(n)
            {
                var ans = 1;
                for (let i = 1; i <= n; i++)
                    ans = ans * i;
                console.log(ans);
            };
            fact1(4);

            Output=>24
